Base Library:
This example demonstrates how to populate a grid from an HTML table. In this case, the table whose data is to be loaded into the grid has an id="derby", and the table's body element is passed to the buffer constructor which then loads the data. 
Rico.onLoad( function() {
  var opts = { frozenColumns: 1, highlightElem: 'cursorRow', menuEvent: 'click', visibleRows: 'body', defaultWidth: 120 };
  var buffer=new Rico.Buffer.Base(Rico.$('derby').tBodies[0]);
  var derby=new Rico.LiveGrid ('derby', buffer, opts);
  derby.menu=new Rico.GridMenu();
});
